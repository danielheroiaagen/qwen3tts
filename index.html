# Quick Start with Qwen3-TTS
from qwen_tts import Qwen3TTS

# Initialize the model
model = Qwen3TTS("Qwen/Qwen3-TTS")

# Generate speech
audio = model.synthesize(
    text="Hello world!",
    language="en",
    voice="natural"
)
audio.save("output.wav")
